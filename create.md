def adFunc(a, val):
    for i in range(val -1):
        for j in range(val - i - 1):
            if(a[j] > a[j + 1]):
                temp = a[j]
                a[j] = a[j + 1]
                a[j + 1] = temp

def deFunc(a, val):
    for i in range(val -1):
